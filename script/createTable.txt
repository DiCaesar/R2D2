CREATE TABLE `t_categories` (
  `category_id` int(4) unsigned NOT NULL AUTO_INCREMENT,
  `category_name` varchar(50) NOT NULL,
  `category_desc` varchar(50) DEFAULT NULL,
  `usable_flag` varchar(10) DEFAULT NULL,
  KEY `PK_t_categories_id` (`category_id`)
) ENGINE=InnoDB AUTO_INCREMENT=1499 DEFAULT CHARSET=utf8;



CREATE TABLE `t_category_group_details` (
  `id` int(8) unsigned NOT NULL AUTO_INCREMENT,
  `category_id` int(4) unsigned NOT NULL,
  `group_name` varchar(500) DEFAULT NULL,
  `group_desc` varchar(50) DEFAULT NULL,
  `group_url` varchar(200) DEFAULT NULL,
  `tags` varchar(2500) DEFAULT NULL,
  `pic_list` varchar(5000) DEFAULT NULL,
  `file_path` varchar(50) DEFAULT NULL,
  `download_flag` varchar(10) DEFAULT NULL,
  `usable_flag` varchar(10) DEFAULT NULL,
  UNIQUE KEY `PK_category_details_id` (`id`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=78069 DEFAULT CHARSET=utf8;
